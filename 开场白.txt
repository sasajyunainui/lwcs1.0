function splitSpirits(text) {
    var value = (text || '').trim();
    var list = value ? value.split(/[、,，;；\s]+/) : [];
    list = list.map(function(item) { return item.trim(); }).filter(function(item) { return item !== ''; });
    return list.length ? list.slice(0, 2) : ['未觉醒'];
}

function 获取开场有效天赋(天赋标签, 年龄值) {
    if (年龄值 < 15 && ['天才', '顶级天才', '绝世妖孽'].indexOf(天赋标签) >= 0) return '天才';
    if (年龄值 < 18 && ['顶级天才', '绝世妖孽'].indexOf(天赋标签) >= 0) return '顶级天才';
    return 天赋标签 || '正常';
}

function 插值开场等级(描点列表, 年龄值) {
    var 等级值 = 描点列表[0][1];
    for (var 序号 = 0; 序号 < 描点列表.length - 1; 序号++) {
        var 当前描点 = 描点列表[序号];
        var 下个描点 = 描点列表[序号 + 1];
        if (年龄值 < 当前描点[0]) break;
        if (年龄值 <= 下个描点[0]) {
            var 比例 = Math.max(0, Math.min(1, (年龄值 - 当前描点[0]) / Math.max(1, 下个描点[0] - 当前描点[0])));
            return 当前描点[1] + (下个描点[1] - 当前描点[1]) * 比例;
        }
        等级值 = 下个描点[1];
    }
    return 等级值;
}

function 解析开场生日年内日序(生日) {
    var 文本 = String(生日 || '').trim();
    if (!文本 || 文本 === '待生成') return null;
    var 匹配 = 文本.match(/^(\d{1,2})\s*月\s*(\d{1,2})\s*日?$/) || 文本.match(/^(\d{1,2})\s*[-/.]\s*(\d{1,2})$/);
    if (!匹配) return null;
    var 月 = Math.max(1, Math.min(12, Math.floor(Number(匹配[1]) || 1)));
    var 日 = Math.max(1, Math.min(30, Math.floor(Number(匹配[2]) || 1)));
    return (月 - 1) * 30 + (日 - 1);
}

function 格式化开场生日(月, 日) {
    var 月值 = Math.max(1, Math.min(12, Math.floor(Number(月) || 1)));
    var 日值 = Math.max(1, Math.min(30, Math.floor(Number(日) || 1)));
    return 月值 + '月' + 日值 + '日';
}

function 读取开场生日() {
    var monthEl = document.getElementById('char-birthday-month');
    var dayEl = document.getElementById('char-birthday-day');
    return 格式化开场生日(monthEl && monthEl.value, dayEl && dayEl.value);
}

function 写入开场生日(生日) {
    var monthEl = document.getElementById('char-birthday-month');
    var dayEl = document.getElementById('char-birthday-day');
    if (!monthEl || !dayEl) return;
    var 匹配 = String(生日 || '').trim().match(/^(\d{1,2})\s*月\s*(\d{1,2})\s*日?$/) || String(生日 || '').trim().match(/^(\d{1,2})\s*[-/.]\s*(\d{1,2})$/);
    monthEl.value = String(Math.max(1, Math.min(12, Math.floor(Number(匹配 && 匹配[1]) || 1))));
    dayEl.value = String(Math.max(1, Math.min(30, Math.floor(Number(匹配 && 匹配[2]) || 1))));
}

function 初始化开场生日选项() {
    var monthEl = document.getElementById('char-birthday-month');
    var dayEl = document.getElementById('char-birthday-day');
    if (!monthEl || !dayEl) return;
    if (!monthEl.options.length) {
        for (var 月 = 1; 月 <= 12; 月++) {
            monthEl.appendChild(new Option(月 + '月', String(月)));
        }
    }
    if (!dayEl.options.length) {
        for (var 日 = 1; 日 <= 30; 日++) {
            dayEl.appendChild(new Option(日 + '日', String(日)));
        }
    }
    写入开场生日(读取开场生日());
}

function 计算开场tick年内日序(tick) {
    var 总天数 = Math.floor(Math.max(0, Number(tick || 0)) / 144);
    return ((总天数 % 360) + 360) % 360;
}

function 计算开场生日有效年龄(年龄值, 生日, 当前tick) {
    var 基础年龄 = Math.max(0, Number(年龄值) || 0);
    var 生日序 = 解析开场生日年内日序(生日);
    if (生日序 === null) return 基础年龄;
    var 当前序 = 计算开场tick年内日序(当前tick);
    var 距上次生日天数 = (当前序 - 生日序 + 360) % 360;
    return 基础年龄 + 距上次生日天数 / 360;
}

function 获取开场系别修炼难度系数(系别) {
    if (typeof 系别 === 'number') return Math.max(0.1, Number(系别) || 1);
    var 系别修炼难度表 = {
        '强攻系': 1,
        '防御系': 1,
        '敏攻系': 1,
        '控制系': 1,
        '辅助系': 1.2,
        '治疗系': 1.2,
        '食物系': 1.2,
        '精神系': 1,
        '元素系': 1
    };
    return Math.max(0.1, Number(系别修炼难度表[String(系别 || '强攻系').trim()] || 1));
}

function 获取开场双生修炼难度系数(系别列表) {
    var 有效列表 = Array.isArray(系别列表) && 系别列表.length ? 系别列表 : ['强攻系'];
    var 最低倍率 = Infinity;
    有效列表.forEach(function(系别) {
        最低倍率 = Math.min(最低倍率, 获取开场系别修炼难度系数(系别));
    });
    return Math.max(0.1, Number.isFinite(最低倍率) ? 最低倍率 : 1);
}

function 规避开场魂环门槛等级(等级值) {
    if (等级值 >= 99.5) return 99.5;
    if (等级值 >= 10 && 等级值 <= 90 && 等级值 % 10 === 0) return 等级值 + 1;
    return 等级值;
}

function 计算开场基础魂力上限(等级值) {
    var 等级 = Math.max(1, Math.min(99.5, Number(等级值) || 1));
    if (等级 <= 29) return 100 + ((2200 - 100) / 28) * (等级 - 1);
    if (等级 === 30) return 3000;
    if (等级 <= 69) return 3200 + ((9000 - 3200) / 38) * (等级 - 31);
    if (等级 === 70) return 14000;
    if (等级 <= 89) return 14500 + ((17000 - 14500) / 18) * (等级 - 71);
    if (等级 === 90) return 18500;
    if (等级 <= 94) return 18875 + ((20000 - 18875) / 3) * (等级 - 91);
    if (等级 <= 99) return 20000 * Math.pow(2, 等级 - 94);
    return 20000 * Math.pow(2, 99 - 94) * 2;
}

function 按开场系别折算等级(等级值, 系别) {
    var 系别倍率 = 获取开场系别修炼难度系数(系别);
    if (系别倍率 <= 1) return 等级值;
    var 同等修炼魂力 = 计算开场基础魂力上限(等级值);
    var 折算等级 = 1;
    for (var 候选等级 = 1; 候选等级 <= 99.5; 候选等级 += 0.5) {
        if (计算开场基础魂力上限(候选等级) * 系别倍率 <= 同等修炼魂力) {
            折算等级 = 候选等级;
        } else {
            break;
        }
    }
    return Math.min(等级值, 折算等级);
}

function 计算开场等级(天赋配置, 年龄值, 系别, 生日, 当前tick) {
    var 有效年龄 = 计算开场生日有效年龄(年龄值, 生日, 当前tick);
    var 有效天赋 = 获取开场有效天赋(天赋配置.label, 有效年龄);
    var 等级描点表 = {
        '劣等': [[6, 1], [10, 1], [20, 6], [30, 14], [40, 21], [150, 21]],
        '正常': [[6, 3], [10, 6], [20, 21], [30, 32], [50, 41], [150, 41]],
        '优秀': [[6, 6], [10, 14], [20, 40], [30, 69], [60, 70], [150, 70]],
        '天才': [[6, 8], [9, 20], [10, 21], [20, 48], [30, 69], [40, 80], [50, 85], [100, 90], [150, 90]],
        '顶级天才': [[6, 8], [10, 21], [20, 53], [30, 74], [40, 89], [50, 94], [100, 96], [150, 96]],
        '绝世妖孽': [[6, 8], [10, 21], [20, 54], [30, 98], [40, 99.5], [150, 99.5]]
    };
    var 等级值 = 插值开场等级(等级描点表[有效天赋] || 等级描点表['正常'], Math.max(0, Number(有效年龄) || 6));
    var 折算等级 = 按开场系别折算等级(等级值, 系别);
    var 整理等级 = 折算等级 >= 99.5 ? 99.5 : Math.floor(折算等级);
    return Math.max(1, Math.min(99.5, 规避开场魂环门槛等级(整理等级)));
}

function readStorageJSON(key) {
    try {
        var raw = localStorage.getItem(key);
        if (!raw) return {};
        var parsed = JSON.parse(raw);
        return (parsed && typeof parsed === 'object') ? parsed : {};
    } catch (err) {
        return {};
    }
}

function writeStorageJSON(key, value) {
    try {
        localStorage.setItem(key, JSON.stringify(value || {}));
        return true;
    } catch (err) {
        return false;
    }
}

function refreshTemplateSelect(selectedName) {
    var select = document.getElementById('template-select');
    if (!select) return;

    var store = loadTemplateStore();
    var names = Object.keys(store).sort(function(a, b) {
        var at = (store[a] && store[a].updatedAt) || 0;
        var bt = (store[b] && store[b].updatedAt) || 0;
        if (bt !== at) return bt - at;
        return a > b ? 1 : (a < b ? -1 : 0);
    });

    select.innerHTML = '';
    var placeholder = document.createElement('option');
    placeholder.value = '';
    placeholder.textContent = '选择已保存模板';
    select.appendChild(placeholder);

    names.forEach(function(name) {
        var option = document.createElement('option');
        option.value = name;
        option.textContent = name;
        select.appendChild(option);
    });

    if (selectedName && store[selectedName]) {
        select.value = selectedName;
    }
}

}

window.saveCurrentTemplate = function() {
    var nameInput = document.getElementById('template-name');
    var name = (nameInput && nameInput.value ? nameInput.value : '').trim();
    if (!name) {
        showTemplateMessage('请先输入模板名，再点击保存。', true);
        return;
    }

    var store = loadTemplateStore();
    store[name] = {
        updatedAt: Date.now(),
        snapshot: collectFormSnapshot()
    };

    if (!saveTemplateStore(store)) {
        showTemplateMessage('保存失败：本地存储不可用。', true);
        return;
    }

    refreshTemplateSelect(name);
    showTemplateMessage('已保存模板：' + name);
};

window.loadSelectedTemplate = function() {
    var select = document.getElementById('template-select');
    var name = (select && select.value) ? select.value : '';
    if (!name) {
        showTemplateMessage('请先从列表选择模板。', true);
        return;
    }

    var store = loadTemplateStore();
    var entry = store[name];
    if (!entry || !entry.snapshot) {
        showTemplateMessage('模板不存在，列表已刷新。', true);
        refreshTemplateSelect();
        return;
    }

    applyFormSnapshot(entry.snapshot);
    saveLastSnapshot(entry.snapshot);

    var nameInput = document.getElementById('template-name');
    if (nameInput) nameInput.value = name;

    showTemplateMessage('已加载模板：' + name);
};

window.deleteSelectedTemplate = function() {
    var select = document.getElementById('template-select');
    var name = (select && select.value) ? select.value : '';
    if (!name) {
        showTemplateMessage('请先选择要删除的模板。', true);
        return;
    }

    var store = loadTemplateStore();
    if (!store[name]) {
        showTemplateMessage('模板不存在，列表已刷新。', true);
        refreshTemplateSelect();
        return;
    }

    delete store[name];
    if (!saveTemplateStore(store)) {
        showTemplateMessage('删除失败：本地存储不可用。', true);
        return;
    }

    refreshTemplateSelect();
    showTemplateMessage('已删除模板：' + name);
};

    async function applyInitPatchByMvuParser(patchList) {
        var runtimeWindow = getStartupRuntimeWindow();
        var mvu = (runtimeWindow && runtimeWindow.Mvu) || window.Mvu;
        if (!mvu || typeof mvu.getMvuData !== "function" || typeof mvu.replaceMvuData !== "function" || typeof mvu.parseMessage !== "function") {
            throw new Error("MVU更新解析接口未就绪");
        }
        var currentData = await mvu.getMvuData({ type: "message", message_id: "latest" });
        var parsedData = await Promise.resolve(mvu.parseMessage(buildInitUpdateVariableText(patchList), currentData));
        var nextData = parsedData && parsedData.stat_data ? parsedData : currentData;
        if (!nextData || !nextData.stat_data) throw new Error("MVU初始化解析失败");
        var normalizeStatData = runtimeWindow.__LWCS_NORMALIZE_MVU_STAT_DATA__ || window.__LWCS_NORMALIZE_MVU_STAT_DATA__;
        if (typeof normalizeStatData !== "function") throw new Error("MVU结构规范化接口未就绪");
        nextData.stat_data = normalizeStatData(nextData.stat_data);
        delete nextData.stat_data.display_all;
        delete nextData.stat_data.display_chars;
        await mvu.replaceMvuData(nextData, { type: "message", message_id: "latest" });
        await verifyInitPatchLatest();
    }

            throw new Error("MVU初始化校验失败：角色位置未写入最新楼层");
        }
